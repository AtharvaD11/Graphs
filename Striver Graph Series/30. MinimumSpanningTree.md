### Solution : Prims

```
class Solution
{
	public:
	//Function to find sum of weights of edges of the Minimum Spanning Tree.
    int spanningTree(int V, vector<vector<int>> adj[]){
        priority_queue<pair<int,int>> pq;
        vector<int> vis(V, 0);
        pq.push({0, 0});
        
        int sum = 0;
        while(pq.size()){
            auto it = pq.top(); pq.pop();
            
            int node = it.second;
            int wt = it.first*-1;
            
            if(vis[node]) continue;
            vis[node] = 1;
            sum += wt;
            
            for(auto nbr : adj[node]){
                int v = nbr[0];
                int edW = nbr[1];
                if(!vis[v]){
                    pq.push({edW*-1, v});
                }
            }
        }
        return sum;
    }
};

TC : O(Elog(E))
SC : O(V + E)
```
